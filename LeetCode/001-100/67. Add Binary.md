# 67. Add Binary

- Difficulty: Easy.
- Related Topics: Math, String.
- Similar Questions: Add Two Numbers, Multiply Strings, Plus One.

## Problem

Given two binary strings, return their sum (also a binary string).

The input strings are both **non-empty** and contains only characters ```1``` orÂ ```0```.

**Example 1:**

```
Input: a = "11", b = "1"
Output: "100"
```

**Example 2:**

```
Input: a = "1010", b = "1011"
Output: "10101"
```

## Solution 1
**88 ms	41.2 MB**
```javascript
/**
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
let addBinary = function (a, b) {
  let num = BigInt(a) + BigInt(b);
  let str = num.toString();

  let res = '';
  let carryNum = 0;
  for (let i = str.length - 1; i >= 0; i--) {
    if (str[i] === '2') {
      if (carryNum === 1) {
        res = '1' + res;
      } else {
        res = '0' + res;
        carryNum = 1;
      }
    }

    if (str[i] === '1') {
      if (carryNum === 1) {
        res = '0' + res;
      } else {
        res = '1' + res;
      }
    }

    if (str[i] === '0') {
      if (carryNum === 1) {
        res = '1' + res;
        carryNum = 0;
      } else {
        res = '0' + res;
      }
    }
  }

  if (carryNum === 1) {
    res = '1' + res;
  }

  return res;
};
```

/**
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
let addBinary = function (a, b) {
    let result = '';
  let i = a.length - 1;
  let j = b.length - 1;
  let carry = 0;
  while (i >= 0 || j >= 0) {
    let sum = carry;
    if (i >= 0) {
      sum += a[i--] - '0';
    }
    if (j >= 0) {
      sum += b[j--] - '0';
    }
    result = (sum % 2) + result;
    carry = parseInt(sum / 2);
  }
  if (carry > 0) {
    result = 1 + result;
  }
  return result;
};
